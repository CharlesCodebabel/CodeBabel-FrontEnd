import requests

#_input get url
fvc = str(input('URL Favicon site : '))
img_url = fvc

#__requests > image data
img_data = requests.get(img_url).content
with open('favicon.jpg', 'wb') as handler:
    handler.write(img_data)    #write data
    print(f'\nURL: {img_url}') #print url
    print('Download sucess!')
